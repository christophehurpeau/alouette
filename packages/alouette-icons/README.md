## Install

```bash
npm install --save alouette alouette-icons
```

## Usage

```tsx
import { Icon } from "alouette";
import { ArrowLeftRegularIcon } from "alouette-icons/phosphor-icons";

function Component() {
  return <Icon icon={<ArrowLeftRegularIcon />} />;
}
```
